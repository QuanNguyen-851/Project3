create table pd_product
(
    id            bigserial
        constraint p_product_pk
            primary key,
    parent_id     bigint,
    code          varchar(255)
        constraint product_code_unique
            unique,
    name          varchar(255),
    color         varchar(255),
    properties    varchar(255),
    description   text,
    category_id   bigint,
    production_id bigint,
    sale_price    bigint,
    import_price  bigint,
    quantity      bigint,
    status        varchar(255),
    created_date  timestamp default CURRENT_TIMESTAMP not null,
    modified_date timestamp,
    created_by    bigint,
    modified_by   bigint,
    avatar_url    text,
    warranty      bigint    default 0
)
